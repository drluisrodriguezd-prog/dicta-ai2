## 3. SCHEMA SUPABASE (implementar exactamente)

Ejecutar este SQL en Supabase antes de conectar la app:

```sql
-- Tabla de médicos (extiende auth.users)
create table medicos (
  id              uuid primary key references auth.users(id) on delete cascade,
  nombre_completo text not null,
  especialidad    text,
  matricula       text,
  firma_base64    text,
  sello_texto     text,
  preferencias    jsonb default '{
    "estilo_nota": "detallado",
    "formato_impresion": "A4",
    "ultimo_correlativo_receta": 0,
    "plantilla_favorita": "soap",
    "tono_redaccion": "formal"
  }'::jsonb,
  created_at      timestamptz default now(),
  updated_at      timestamptz default now()
);

-- Tabla de pacientes
create table pacientes (
  id                   uuid primary key default gen_random_uuid(),
  medico_id            uuid references medicos(id) on delete cascade,
  nombre_completo      text not null,
  fecha_nacimiento     date,
  sexo                 text check (sexo in ('masculino','femenino','otro')),
  documento_identidad  text,
  telefono             text,
  direccion            text,
  contacto_emergencia  jsonb default '{}'::jsonb,
  alergias             jsonb default '[]'::jsonb,
  medicacion_habitual  jsonb default '[]'::jsonb,
  antecedentes         jsonb default '{}'::jsonb,
  created_at           timestamptz default now(),
  updated_at           timestamptz default now()
);

-- Tabla de episodios
create table episodios (
  id              uuid primary key default gen_random_uuid(),
  codigo          text unique,
  paciente_id     uuid references pacientes(id),
  medico_id       uuid references medicos(id),
  tipo_atencion   text check (tipo_atencion in (
                    'consulta_externa','hospitalizacion',
                    'urgencias','terapia_intensiva')),
  fecha_atencion  date not null default current_date,
  hora_atencion   time,
  servicio        text,
  cama            text,
  estado          text default 'borrador' check (estado in (
                    'borrador','en_revision','validado','cerrado')),
  created_at      timestamptz default now(),
  updated_at      timestamptz default now()
);

-- Tabla de notas clínicas
create table notas_clinicas (
  id              uuid primary key default gen_random_uuid(),
  episodio_id     uuid references episodios(id),
  medico_id       uuid references medicos(id),
  tipo_nota       text check (tipo_nota in (
                    'soap','evolucion','ingreso','consulta_externa',
                    'emergencia','interconsulta','alta','uti',
                    'por_problemas','epicrisis')),
  contenido       jsonb not null default '{}'::jsonb,
  version         integer default 1,
  estado          text default 'borrador' check (estado in (
                    'borrador','validado','cerrado')),
  validado_en     timestamptz,
  validado_por    uuid references medicos(id),
  created_at      timestamptz default now(),
  updated_at      timestamptz default now()
);

-- Tabla de transcripciones
create table transcripciones (
  id                     uuid primary key default gen_random_uuid(),
  nota_id                uuid references notas_clinicas(id),
  seccion                text,
  texto_original         text,
  hablante               text check (hablante in ('paciente','medico','mixto')),
  texto_medico_formal    text,
  version_elegida        integer default 1,
  versiones_ia           jsonb default '[]'::jsonb,
  etiquetas_trazabilidad jsonb default '[]'::jsonb,
  created_at             timestamptz default now()
);

-- Tabla de diagnósticos
create table diagnosticos (
  id                    uuid primary key default gen_random_uuid(),
  nota_id               uuid references notas_clinicas(id),
  codigo_cie10          text,
  nombre_diagnostico    text not null,
  tipo                  text check (tipo in ('principal','diferencial','descartado')),
  confirmado_por_medico boolean default false,
  created_at            timestamptz default now()
);

-- Tabla de recetas
create table recetas (
  id                   uuid primary key default gen_random_uuid(),
  episodio_id          uuid references episodios(id),
  medico_id            uuid references medicos(id),
  paciente_id          uuid references pacientes(id),
  numero_correlativo   integer,
  medicamentos         jsonb default '[]'::jsonb,
  diagnostico_texto    text,
  formato_impresion    text default 'A4',
  estado               text default 'borrador' check (estado in ('borrador','emitida')),
  validada_por_medico  boolean default false,
  created_at           timestamptz default now(),
  updated_at           timestamptz default now()
);

-- Tabla de medicamentos frecuentes (memoria predictiva)
create table medicamentos_frecuentes (
  id               uuid primary key default gen_random_uuid(),
  medico_id        uuid references medicos(id) on delete cascade,
  nombre_generico  text not null,
  nombre_comercial text,
  presentacion     text,
  dosis_usual      text,
  via              text,
  frecuencia       text,
  duracion_usual   text,
  observaciones    text,
  veces_usado      integer default 1,
  ultima_vez       timestamptz default now(),
  created_at       timestamptz default now()
);

-- Tabla de auditoría
create table auditoria (
  id               uuid primary key default gen_random_uuid(),
  medico_id        uuid references medicos(id),
  tabla_afectada   text,
  registro_id      uuid,
  accion           text check (accion in (
                     'crear','editar','validar',
                     'imprimir','exportar','ver','cerrar')),
  datos_anteriores jsonb,
  datos_nuevos     jsonb,
  created_at       timestamptz default now()
);

-- RLS: cada médico solo ve sus datos
alter table medicos              enable row level security;
alter table pacientes            enable row level security;
alter table episodios            enable row level security;
alter table notas_clinicas       enable row level security;
alter table transcripciones      enable row level security;
alter table diagnosticos         enable row level security;
alter table recetas              enable row level security;
alter table medicamentos_frecuentes enable row level security;
alter table auditoria            enable row level security;

create policy "acceso_propio" on medicos
  for all using (id = auth.uid());
create policy "acceso_propio" on pacientes
  for all using (medico_id = auth.uid());
create policy "acceso_propio" on episodios
  for all using (medico_id = auth.uid());
create policy "acceso_propio" on notas_clinicas
  for all using (medico_id = auth.uid());
create policy "acceso_propio" on recetas
  for all using (medico_id = auth.uid());
create policy "acceso_propio" on medicamentos_frecuentes
  for all using (medico_id = auth.uid());
create policy "acceso_propio" on auditoria
  for all using (medico_id = auth.uid());
-- transcripciones y diagnósticos heredan por join con notas_clinicas
```

### BARRA DE CONTROLES DE GRABACIÓN (fija en header de secciones 1 y 2)

```
[● GRABAR] [⏸ PAUSAR] [▶ REANUDAR] [⏹ DETENER] [✎ EDITAR] [✦ REGENERAR IA]
```

**Toggle de hablante (grande, prominente):**
```
[ 👤 PACIENTE ]  ←→  [ 👨‍⚕️ MÉDICO ]
```
- Color verde cuando está en modo PACIENTE
- Color azul cuando está en modo MÉDICO
- Solo el texto del modo PACIENTE se envía a traducción clínica
- El texto del modo MÉDICO se guarda como contexto pero NO contamina la narración clínica

### FLUJO COMPLETO DE GRABACIÓN:
1. Médico presiona GRABAR → se activa Web Speech API (`SpeechRecognition`)
2. El texto transcrito aparece en tiempo real en un área de texto con fondo oscuro diferenciado
3. Al presionar DETENER:
   a. Se guarda `texto_original` con etiqueta del hablante
   b. Si hablante = 'paciente': se envía automáticamente a Claude API para traducción
   c. Claude devuelve 3 versiones de redacción médica formal
   d. Se muestran las 3 versiones en tarjetas seleccionables (radio button visual)
   e. Médico elige la versión o edita manualmente
   f. Al confirmar: se guarda en `transcripciones` con etiquetas de trazabilidad

### PROMPT PARA CLAUDE (traducción coloquial → médico formal):
```
Eres un escriba médico clínico especializado en redacción de expedientes en español latinoamericano formal.

TAREA: Traduce el siguiente relato del paciente a lenguaje médico clínico formal para la sección de {SECCION}.

RELATO ORIGINAL DEL PACIENTE:
"{TEXTO_ORIGINAL}"

REGLAS ESTRICTAS:
1. Usa terminología médica precisa (disnea, pirosis, palpitaciones, etc.)
2. Mantén la cronología y secuencia del relato
3. No agregues información que no esté en el relato
4. No hagas diagnósticos
5. Redacta en tercera persona (el paciente refiere...)
6. Genera exactamente 3 versiones de redacción, de más breve a más detallada
7. Responde SOLO en JSON con esta estructura:
{
  "versiones": [
    { "id": 1, "etiqueta": "Conciso", "texto": "..." },
    { "id": 2, "etiqueta": "Estándar", "texto": "..." },
    { "id": 3, "etiqueta": "Detallado", "texto": "..." }
  ],
  "terminos_detectados": ["disnea", "..."],
  "alertas": ["posible alergia mencionada: ..."] // solo si aplica
}

EJEMPLOS DE TRADUCCIÓN:
- "me falta el aire" → "disnea de esfuerzo"
- "me arde el pecho" → "pirosis / dolor urente retroesternal"
- "me late rápido" → "palpitaciones"
- "me duele la cabeza" → "cefalea"
- "estoy muy cansado" → "astenia marcada"
- "me duele al orinar" → "disuria"
```
